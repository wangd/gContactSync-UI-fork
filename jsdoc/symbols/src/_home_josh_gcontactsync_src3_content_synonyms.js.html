 52 /**
 53  * Creates an XMLSerializer to serialize the given XML then create a more
 54  * human-friendly string representation of that XML.
 55  * This is an expensive method of serializing XML but results in the most
 56  * human-friendly string from XML.
 57  * 
 58  * Also see serializeFromText.
 59  *
 60  * @param aXML {XML} The XML to serialize into a human-friendly string.
 61  * @returns {string}  A formatted string of the given XML.
 62  */
 63 com.gContactSync.serialize = function gCS_serialize(aXML) {
 64   if (!aXML)
 65     return "";
 66   try {
 67     var serializer = new XMLSerializer(),
 68         str        = serializer.serializeToString(aXML);
 69     // source: http://developer.mozilla.org/en/E4X#Known_bugs_and_limitations
 70     str = str.replace(/^<\?xml\s+version\s*=\s*(["'])[^\1]+\1[^?]*\?>/, ""); // bug 336551
 71     return XML(str).toXMLString();
 72   }
 73   catch (e) {
 74     com.gContactSync.LOGGER.LOG_WARNING("Error while serializing the following XML: " +
 75                                         aXML, e);
 76   }
 77   return "";
 78 };

 80 /**
 81  * A less expensive (but still costly) function that serializes a string of XML
 82  * adding newlines between adjacent tags (...><...).
 83  * If the verboseLog preference is set as false then this function does nothing.
 84  *
 85  * @param aString {string} The XML string to serialize.
 86  * @param aForce {boolean} Set to true to force a serialization regardless of
 87  *                         verboseLog.
 88  * @returns {string} The serialized text if verboseLog is true; else the original
 89  *                  text.
 90  */
 91 com.gContactSync.serializeFromText = function gCS_serializeFromText(aString, aForce) {
 92   // if verbose logging is disabled, don't replace >< with >\n< because it only
 93   // wastes time
 94   if (aForce || com.gContactSync.Preferences.mSyncPrefs.verboseLog.value) {
 95     var arr = aString.split("><");
 96     aString = arr.join(">\n<");
 97   }
 98   return aString;
 99 };

101 /**
102  * Creates a 'dummy' e-mail for the given contact if possible.
103  * The dummy e-mail contains 'nobody' (localized) and '@nowhere.invalid' (not
104  * localized) as well as a string of numbers.  The numbers are the ID from
105  * Google, if any, or a random sequence.  The numbers are fairly unique because
106  * mailing lists require contacts with distinct e-mail addresses otherwise they
107  * fail silently.
108  *
109  * The purpose of the dummy e-mail addresses is to prevent mailing list bugs
110  * relating to contacts without e-mail addresses.
111  *
112  * This function checks the 'dummyEmail' pref and if that pref is set as true
113  * then this function will not set the e-mail unless the ignorePref parameter is
114  * supplied and evaluates to true.
115  *
116  * @param aContact A contact from Thunderbird.  It can be one of the following:
117  *                 TBContact, GContact, or an nsIAbCard (Thunderbird 2 or 3)
118  * @param ignorePref {boolean} Set this as true to ignore the preference
119  *                             disabling dummy e-mail addresses.  Use this in
120  *                             situations where not adding an address would
121  *                             definitely cause problems.
122  * @returns {string} A dummy e-mail address.
123  */
124 com.gContactSync.makeDummyEmail = function gCS_makeDummyEmail(aContact, ignorePref) {
125   if (!aContact) throw "Invalid contact sent to makeDummyEmail";
126   if (!ignorePref && !com.gContactSync.Preferences.mSyncPrefs.dummyEmail.value) {
127     com.gContactSync.LOGGER.VERBOSE_LOG(" * Not setting dummy e-mail");
128     return "";
129   }
130   var prefix = com.gContactSync.StringBundle.getStr("dummy1"),
131       suffix = com.gContactSync.StringBundle.getStr("dummy2"),
132       id     = null;
133   // GContact and TBContact may not be defined
134   try {
135     if (aContact instanceof com.gContactSync.GContact)
136       id = aContact.getID(true);
137     // otherwise it is from Thunderbird, so try to get the Google ID, if any
138     else if (aContact instanceof com.gContactSync.TBContact)
139       id = aContact.getID();
140     else
141       id = com.gContactSync.GAbManager.getCardValue(aContact, "GoogleID");
142   } catch (e) {
143     try {
144       // try getting the card's value
145       if (aContact.getProperty) // post Bug 413260
146         id = aContact.getProperty("GoogleID", null);
147       else // pre Bug 413260
148         id = aContact.getStringAttribute("GoogleID");
149     }
150     catch (ex) {}
151   }
152   if (id) {
153     // take just the ID and not the whole URL
154     return prefix + id.substr(1 + id.lastIndexOf("/")) + suffix;
155   }
156   // if there is no ID make a random number and remove the "0."
157   else {
158     return prefix + String(Math.random()).replace("0.", "") + suffix;
159   }
160 };

174 /**
175  * Selects the menuitem with the given value (value or label attribute) in the
176  * given menulist.
177  * Optionally creates the menuitem if it cannot be found.
178  *
179  * @param aMenuList {menulist} The menu list element to search.
180  * @param aValue    {string}   The value to find in a menuitem.  This can be
181  *                             either the 'value' or 'label' attribute of the
182  *                             matched item.  Case insensitive.
183  * @param aCreate   {boolean}  Set as true to create and select a new menuitem
184  *                             if a match cannot be found.
185  */
186 com.gContactSync.selectMenuItem = function gCS_selectMenuItem(aMenuList, aValue, aCreate) {
187   if (!aMenuList || !aMenuList.menupopup || !aValue)
188     throw "Invalid parameter sent to selectMenuItem";
189 
190   var arr = aMenuList.menupopup.childNodes,
191       i,
192       item,
193       aValueLC = aValue.toLowerCase();
194   for (i = 0; i < arr.length; i++) {
195     item = arr[i];
196     if (item.getAttribute("value").toLowerCase() === aValueLC ||
197         item.getAttribute("label").toLowerCase() === aValueLC) {
198       aMenuList.selectedIndex = i;
199       return true;
200     }
201   }
202   if (!aCreate)
203     return false;
204   item = aMenuList.appendItem(aValue, aValue);
205   // getIndexOfItem was added in TB/FF 3
206   aMenuList.selectedIndex = aMenuList.menupopup.childNodes.length - 1;
207   return true;
208 };

210 /**
211  * Attempts a few basic fixes for 'broken' usernames.
212  * In the past, gContactSync didn't check that a username included the domain
213  * which would pass authentication and then fail to do anything else.
214  * It also didn't make sure there were no spaces in a username which would
215  * also pass authentication and break for everything else.
216  * See Bug 21567
217  *
218  * @param aUsername {string} The username to fix.
219  *
220  * @returns {string} A username with a domain and no spaces.
221  */
222 com.gContactSync.fixUsername = function gCS_fixUsername(aUsername) {
223   if (!aUsername)
224     return null;
225   // Add @gmail.com if necessary
226   if (aUsername.indexOf("@") === -1)
227     aUsername += "@gmail.com";
228   // replace any spaces or tabs
229   aUsername = aUsername.replace(/[ \t\n\r]/g, "");
230   return aUsername;
231 };

325 /**
326  * Opens the given URL using the openFormattedURL and
327  * openFormattedRegionURL functions.
328  *
329  * @param aURL {string} THe URL to open.
330  */
331 com.gContactSync.openURL = function gCS_openURL(aURL) {
332   com.gContactSync.LOGGER.VERBOSE_LOG("Opening the following URL: " + aURL);
333   if (!aURL) {
334     com.gContactSync.LOGGER.LOG_WARNING("Caught an attempt to load a blank URL");
335     return;
336   }
337   try {
338     if (openFormattedURL) {
339       openFormattedURL(aURL);
340       return;
341     }
342   }
343   catch (e) {
344     com.gContactSync.LOGGER.LOG_WARNING(" - Error in openFormattedURL", e);
345   }
346   try {
347     if (openFormattedRegionURL) {
348       openFormattedRegionURL(aURL);
349       return;
350     }
351   }
352   catch (e) {
353     com.gContactSync.LOGGER.LOG_WARNING(" - Error in openFormattedURL", e);
354   }
355   com.gContactSync.LOGGER.LOG_WARNING("Could not open the URL: " + aURL);
356   return;
357 };

373 /**
374  * Replaces https://... with http://... in URLs as a permanent workaround for
375  * the issue described here:
376  * http://www.google.com/support/forum/p/apps-apis/thread?tid=6fde249ce2ffe7a9&hl=en
377  *
378  * @param aURL {string} The URL to fix.
379  * @return {string} The URL using https instead of http
380  */
381 com.gContactSync.fixURL = function gCS_fixURL(aURL) {
382   if (!aURL) {
383     return aURL;
384   }
385   return aURL.replace(/^https:/i, "http:");
386 };
