 52 /**
 53  * Creates an XMLSerializer to serialize the given XML then create a more
 54  * human-friendly string representation of that XML.
 55  * This is an expensive method of serializing XML but results in the most
 56  * human-friendly string from XML.
 57  * 
 58  * Also see serializeFromText.
 59  *
 60  * @param aXML {XML} The XML to serialize into a human-friendly string.
 61  * @returns {string}  A formatted string of the given XML.
 62  */
 63 com.gContactSync.serialize = function gCS_serialize(aXML) {
 64   if (!aXML)
 65     return "";
 66   try {
 67     var serializer = new XMLSerializer(),
 68         str        = serializer.serializeToString(aXML);
 69     // source: http://developer.mozilla.org/en/E4X#Known_bugs_and_limitations
 70     str = str.replace(/^<\?xml\s+version\s*=\s*(["'])[^\1]+\1[^?]*\?>/, ""); // bug 336551
 71     return XML(str).toXMLString();
 72   }
 73   catch (e) {
 74     com.gContactSync.LOGGER.LOG_WARNING("Error while serializing the following XML: " +
 75                                         aXML, e);
 76   }
 77   return "";
 78 };

 80 /**
 81  * A less expensive (but still costly) function that serializes a string of XML
 82  * adding newlines between adjacent tags (...><...).
 83  * If the verboseLog preference is set as false then this function does nothing.
 84  *
 85  * @param aString {string} The XML string to serialize.
 86  * @returns {string} The serialized text if verboseLog is true; else the original
 87  *                  text.
 88  */
 89 com.gContactSync.serializeFromText = function gCS_serializeFromText(aString) {
 90   // if verbose logging is disabled, don't replace >< with >\n< because it only
 91   // wastes time
 92   if (com.gContactSync.Preferences.mSyncPrefs.verboseLog.value) {
 93     var arr = aString.split("><");
 94     aString = arr.join(">\n<");
 95   }
 96   return aString;
 97 };

 99 /**
100  * Creates a 'dummy' e-mail for the given contact if possible.
101  * The dummy e-mail contains 'nobody' (localized) and '@nowhere.invalid' (not
102  * localized) as well as a string of numbers.  The numbers are the ID from
103  * Google, if any, or a random sequence.  The numbers are fairly unique because
104  * mailing lists require contacts with distinct e-mail addresses otherwise they
105  * fail silently.
106  *
107  * The purpose of the dummy e-mail addresses is to prevent mailing list bugs
108  * relating to contacts without e-mail addresses.
109  *
110  * This function checks the 'dummyEmail' pref and if that pref is set as true
111  * then this function will not set the e-mail unless the ignorePref parameter is
112  * supplied and evaluates to true.
113  *
114  * @param aContact A contact from Thunderbird.  It can be one of the following:
115  *                 TBContact, GContact, or an nsIAbCard (Thunderbird 2 or 3)
116  * @param ignorePref {boolean} Set this as true to ignore the preference
117  *                             disabling dummy e-mail addresses.  Use this in
118  *                             situations where not adding an address would
119  *                             definitely cause problems.
120  * @returns {string} A dummy e-mail address.
121  */
122 com.gContactSync.makeDummyEmail = function gCS_makeDummyEmail(aContact, ignorePref) {
123   if (!aContact) throw "Invalid contact sent to makeDummyEmail";
124   if (!ignorePref && !com.gContactSync.Preferences.mSyncPrefs.dummyEmail.value) {
125     com.gContactSync.LOGGER.VERBOSE_LOG(" * Not setting dummy e-mail");
126     return "";
127   }
128   var prefix = com.gContactSync.StringBundle.getStr("dummy1"),
129       suffix = com.gContactSync.StringBundle.getStr("dummy2"),
130       id     = null;
131   // GContact and TBContact may not be defined
132   try {
133     if (aContact instanceof com.gContactSync.GContact)
134       id = aContact.getID(true);
135     // otherwise it is from Thunderbird, so try to get the Google ID, if any
136     else if (aContact instanceof com.gContactSync.TBContact)
137       id = aContact.getValue("GoogleID");
138     else
139       id = com.gContactSync.GAbManager.getCardValue(aContact, "GoogleID");
140   } catch (e) {
141     try {
142       // try getting the card's value
143       if (aContact.getProperty) // post Bug 413260
144         id = aContact.getProperty("GoogleID", null);
145       else // pre Bug 413260
146         id = aContact.getStringAttribute("GoogleID");
147     }
148     catch (ex) {}
149   }
150   if (id) {
151     // take just the ID and not the whole URL
152     return prefix + id.substr(1 + id.lastIndexOf("/")) + suffix;
153   }
154   // if there is no ID make a random number and remove the "0."
155   else {
156     return prefix + String(Math.random()).replace("0.", "") + suffix;
157   }
158 };

162  *
163  * @param aEmail {string} The e-mail address to check.

172 /**
173  * Selects the menuitem with the given value (value or label attribute) in the
174  * given menulist.
175  * Optionally creates the menuitem if it cannot be found.
176  *
177  * @param aMenuList {menulist} The menu list element to search.
178  * @param aValue    {string}   The value to find in a menuitem.  This can be
179  *                             either the 'value' or 'label' attribute of the
180  *                             matched item.  Case insensitive.
181  * @param aCreate   {boolean}  Set as true to create and select a new menuitem
182  *                             if a match cannot be found.
183  */
184 com.gContactSync.selectMenuItem = function gCS_selectMenuItem(aMenuList, aValue, aCreate) {
185   if (!aMenuList || !aMenuList.menupopup || !aValue)
186     throw "Invalid parameter sent to selectMenuItem";
187 
188   var arr = aMenuList.menupopup.childNodes,
189       i,
190       item,
191       aValueLC = aValue.toLowerCase();
192   for (i = 0; i < arr.length; i++) {
193     item = arr[i];
194     if (item.getAttribute("value").toLowerCase() === aValueLC ||
195         item.getAttribute("label").toLowerCase() === aValueLC) {
196       aMenuList.selectedIndex = i;
197       return true;
198     }
199   }
200   if (!aCreate)
201     return false;
202   item = aMenuList.appendItem(aValue, aValue);
203   // getIndexOfItem was added in TB/FF 3
204   aMenuList.selectedIndex = aMenuList.menupopup.childNodes.length - 1;
205   return true;
206 };

208 /**
209  * Attempts a few basic fixes for 'broken' usernames.
210  * In the past, gContactSync didn't check that a username included the domain
211  * which would pass authentication and then fail to do anything else.
212  * It also didn't make sure there were no spaces in a username which would
213  * also pass authentication and break for everything else.
214  * See Bug 21567
215  *
216  * @param aUsername {string} The username to fix.
217  *
218  * @returns {string} A username with a domain and no spaces.
219  */
220 com.gContactSync.fixUsername = function gCS_fixUsername(aUsername) {
221   if (!aUsername)
222     return null;
223   // Add @gmail.com if necessary
224   if (aUsername.indexOf("@") === -1)
225     aUsername += "@gmail.com";
226   // replace any spaces or tabs
227   aUsername = aUsername.replace(/[ \t\n\r]/g, "");
228   return aUsername;
229 };
