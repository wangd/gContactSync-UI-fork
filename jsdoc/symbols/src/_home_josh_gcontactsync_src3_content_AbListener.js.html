 37 if (!com) var com = {}; // A generic wrapper variable
 38 // A wrapper for all GCS functions and variables
 39 if (!com.gContactSync) com.gContactSync = {};

 41 /**
 42  * AbListener is a listener for the Address Book that is currently only used to
 43  * update the last modified date of mailing lists and cards contained within
 44  * them.
 45  * @class
 46  */
 47 com.gContactSync.AbListener = {
 48   /**
 49    * Unused.
 50    * @param aParentDir The parent directory to which an item was added.
 51    * @param aItem      The item added to the directory.
 52    */
 53   onItemAdded: function AbListener_onItemAdded(aParentDir, aItem) {},
 54   /**
 55    * Unused.
 56    * @param aItem     The item whose property was changed.
 57    * @param aProperty The property changed.
 58    * @param aOldValue The former value of the property.
 59    * @param aNewValue The new value of the property.
 60    */
 61   onItemPropertyChanged: function AbListener_onItemPropertyChanged(aItem,
 62                                                                    aProperty,
 63                                                                    aOldValue,
 64                                                                    aNewValue) {},
 65   /**
 66    * Used just to update the lastModifiedDate of cards removed from a mail list.
 67    * If a mail list is removed nothing needs to be done since the Group will be
 68    * deleted in Gmail.
 69    * @param aParentDir The directory from which an item was removed.  Ignored
 70    *                   unless it is a mail list.
 71    * @param aItem      The item removed from a directory.  Ignored unless it is
 72    *                   an Address Book card removed from a mail list.
 73    */
 74   onItemRemoved: function AbListener_onItemRemoved(aParentDir, aItem) {
 75     aParentDir.QueryInterface(Components.interfaces.nsIAbDirectory);
 76 
 77     // if a contact was removed and there is not an ongoing synchronization
 78     if (aItem instanceof Components.interfaces.nsIAbCard && com.gContactSync.Sync.mSynced) {
 79       // if a contact was removed from just a amiling list, update the contact's
 80       // last modified date in the parent address book
 81       if (aParentDir.isMailList) {
 82         try {
 83           aItem.QueryInterface(Components.interfaces.nsIAbCard);
 84           var now     = (new Date()).getTime() / 1000,
 85               uri     = this.getURI(aParentDir),
 86               // the parent of aParentDir (aParentDir is a mailing list, dir is the
 87               // directory in which aParentDir is stored)
 88               dir     = new com.gContactSync.GAddressBook(com.gContactSync.AbManager.getAbByURI(uri)),
 89               contact = new com.gContactSync.TBContact(aItem, dir);
 90           // the URI of the list's parent
 91           uri = uri.substring(0, uri.lastIndexOf("/"));
 92           // set the last modified date and update the card
 93           contact.setValue("LastModifiedDate", now);
 94           contact.update();
 95         }
 96         catch (e) {
 97           com.gContactSync.LOGGER.LOG_WARNING("Error updating card after being removed: " + 
 98                              aItem + " " + uri + " " + now, e);
 99         }
100       }
101     }
102   },
103   /**
104    * Gets the Uniform Resource Identifier (URI) of the specified directory.
105    * @param aDirectory {nsIAbDirectory} The directory whose URI is returned.
106    * @returns {string} The URI of aDirectory.
107    */
108   getURI: function AbListener_getURI(aDirectory) {
109     var error;
110     try {
111       if (aDirectory && (aDirectory instanceof Components.interfaces.nsIAbDirectory)) {
112         if (aDirectory.URI) // Thunderbird 3
113           return aDirectory.URI;
114         aDirectory.QueryInterface(Components.interfaces.nsIAbMDBDirectory); // Thunderbird 2
115         if (aDirectory.getDirUri)
116           return aDirectory.getDirUri();
117      } 
118     } catch (e) { error = e; }
119     com.gContactSync.LOGGER.LOG_WARNING("AbListener could not get a URI for: " + aDirectory,
120                        error);
121     return "";
122   },
123   /**
124    * Adds this listener to be alerted whenever a directory item is removed.
125    * It will be called whenever an item (card or mail list) is removed from a
126    * directory (address book or mail list).
127    */
128   add: function AbListener_add() {
129     var flags;
130     if (Components.classes["@mozilla.org/abmanager;1"]) { // Thunderbird 3
131       flags = Components.interfaces.nsIAbListener.directoryItemRemoved;
132       Components.classes["@mozilla.org/abmanager;1"]
133                 .getService(Components.interfaces.nsIAbManager)
134                 .addAddressBookListener(com.gContactSync.AbListener, flags);
135     }
136     else { // Thunderbird 2
137       flags = Components.interfaces.nsIAddrBookSession.directoryItemRemoved;
138       Components.classes["@mozilla.org/addressbook/services/session;1"]
139                 .getService(Components.interfaces.nsIAddrBookSession)
140                 .addAddressBookListener(com.gContactSync.AbListener, flags);
141     }
142   },
143   /**
144    * AbListener.remove
145    * Removes this listener.
146    */
147   remove: function AbListener_remove() {
148     if (Components.classes["@mozilla.org/abmanager;1"]) // Thunderbird 3
149       Components.classes["@mozilla.org/abmanager;1"]
150                 .getService(Components.interfaces.nsIAbManager)
151                 .removeAddressBookListener(com.gContactSync.AbListener);
152     else // Thunderbird 2
153       Components.classes["@mozilla.org/addressbook/services/session;1"]
154                 .getService(Components.interfaces.nsIAddrBookSession)
155                 .removeAddressBookListener(com.gContactSync.AbListener);
156   }
157 };
