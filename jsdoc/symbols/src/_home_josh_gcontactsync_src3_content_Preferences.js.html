 87   /**
 88    * Registers the pref observer and gets the initial preference values.

 90   register: function CP_Preferences_register() {
 91     // Add an observer

100       return;
101     }

103     this.mSyncBranch.removeObserver("", this);
104     this.mRegistered = false;

106   /**
107    * Called when a preference changes on the extensions.gContactSync. branch.
108    *
109    * @param aSubject {nsIPrefBranch} The branch.
110    * @param aTopic {string} A description of what happened.
111    * @param aData  {string} The name of the pref that was changed.
112    */
113   observe: function(aSubject, aTopic, aData) {
114     if (aTopic != "nsPref:changed") {
115       return;
116     }
117     // TODO - determine the cause of 'com is not defined' errors
118     // this observer shouldn't be registered when com isn't defined.
119     try {
120       com.gContactSync.LOGGER.VERBOSE_LOG("**Observed a preference change: " + aData + " - " + aTopic);
121     }
122     catch (e) {
123       return;
124     }
125     var pref = this.mSyncPrefs[aData];
126     if (pref) {
127       var oldValue = pref.value;
128       pref.value = this.getPref(this.mSyncBranch, pref.label, pref.type);
129       com.gContactSync.LOGGER.VERBOSE_LOG(" - Old value: '" + oldValue + "'\n" +
130                                           " - New value: '" + pref.value + "'");
131       switch (aData) {
132       case "statusBarText":
133         var elem = document.getElementById("gContactSyncStatusText");
134         if (elem) {
135           elem.label = pref.value;
136         }
137         break;
138       case "enableMenu":
139         var elem = document.getElementById("gContactSyncMenu");
140         if (elem) {
141           elem.collapsed = !pref.value;
142         }
143         break;
144       }
145     }
146     // if it isn't a sync pref, check if it is a preference for an existing
147     // GAddressBook
148     else {
149       try {
150         var ab = com.gContactSync.GAbManager.mABs[this.mBranchName + aData.substring(0, aData.lastIndexOf(".") + 1)];
151         if (ab) {
152           var pref         = aData.substring(aData.lastIndexOf(".") + 1),
153               prefNoPrefix = pref.replace(ab.prefPrefix, "");
154               newPrefValue = ab.getStringPref(pref);
155           com.gContactSync.LOGGER.VERBOSE_LOG("Changing AB pref: " + pref +
156                                               "\nFrom: " + ab.mPrefs[prefNoPrefix] +
157                                               "\nTo: " + newPrefValue);
158           ab.mPrefs[prefNoPrefix] = newPrefValue;
159         }
160       }
161       catch (ex) {} // ignore errors (GAbManager may not be defined)
162     }
163   },

214   /**
215    * Gets a preference given its branch, name, and type
216    * @param aBranch   {nsIPrefBranch} The branch where the preference is stored.
217    * @param aName     {string} The name of the preference
218    * @param aType     {string} The type of preference.
219    *                           Must be in Preferences.mTypes.
220    */
221   getPref: function Preferences_getPref(aBranch, aName, aType) {
222     if (!aBranch)
223       throw "Invalid aBranch parameter supplied to the getPref method" +
224             com.gContactSync.StringBundle.getStr("pleaseReport");
225     switch (aType) {
226       case this.mTypes.INT:
227         return aBranch.getIntPref(aName);
228       case this.mTypes.BOOL:
229         return aBranch.getBoolPref(aName);
230       case this.mTypes.CHAR:
231         return aBranch.getCharPref(aName);
232       default:
233         throw "Invalid aType parameter supplied to the getPref method" +
234               com.gContactSync.StringBundle.getStr("pleaseReport");
235     }
236   },
237   /**
238    * Sets a preference given its branch, name, type and value.
239    * @param aBranch   {nsIBranch} The branch where the preference is stored.
240    * @param aName     {string}    The name of the preference.
241    * @param aType     {string}    The type of preference.
242    *                              Must be in Preferences.mTypes.
243    * @param aValue    {string}    The value to set the preference.
244    */
245   setPref: function Preferences_setPref(aBranch, aName, aType, aValue) {
246     if (!aBranch)
247       throw "Invalid aBranch parameter supplied to the setPref method" +
248             com.gContactSync.StringBundle.getStr("pleaseReport");
249     switch (aType) {
250       case this.mTypes.INT:
251         return aBranch.setIntPref(aName, aValue);
252       case this.mTypes.BOOL:
253         return aBranch.setBoolPref(aName, aValue);
254       case this.mTypes.CHAR:
255         return aBranch.setCharPref(aName, aValue);
256       default:
257         throw "Invalid aType parameter supplied to the setPref method" +
258               com.gContactSync.StringBundle.getStr("pleaseReport");
259     }
260   },
261   /**
262    * A convienient method of saving a sync preference.
263    * @param aPrefName {string} The preference on the gContactSync branch
264    *                           to save.
265    * @param aValue {string}    The new value for the given preference.
266    */
267   setSyncPref: function Preferences_setSyncPref(aPrefName, aValue) {
268     var pref = this.mSyncPrefs[aPrefName];
269     if (!pref) {
270       throw "Error - invalid pref name '" + aPrefName + "'" +
271             " sent to setSyncPref";
272     }
273     return this.setPref(this.mSyncBranch, pref.label, pref.type, aValue);
274   },
275   /**
276    * Tries to get each preference in mSyncPrefs and creates the preference and
277    * sets its default value if it is not present.
278    */
279   getSyncPrefs: function Preferences_getSyncPrefs() {
280     com.gContactSync.LOGGER.LOG("\n***Loading Preferences***");
281     for (var i in this.mSyncPrefs) {
282       try {
283         this.mSyncPrefs[i].value = this.getPref(this.mSyncBranch,
284                                                 this.mSyncPrefs[i].label,
285                                                 this.mSyncPrefs[i].type);
286       }
287       catch (e) { // if it doesn't exist make it and set the value to its default
288         this.mSyncPrefs[i].value = this.mSyncPrefs[i].defaultValue;
289         this.setPref(this.mSyncBranch, this.mSyncPrefs[i].label,
290                      this.mSyncPrefs[i].type, this.mSyncPrefs[i].defaultValue);
291       }
292       com.gContactSync.LOGGER.LOG(" * " + i + ": " + this.mSyncPrefs[i].value);
293     }
294     com.gContactSync.LOGGER.LOG("***Finished Loading Preferences***\n");
295     // only add these extended properties if the pref to sync them is true
296     this.mExtendedProperties = [];
297     if (this.mSyncPrefs.syncExtended.value)
298       for (var i = 1; i <= 10; i++)
299         this.mExtendedProperties.push(this.getPref(this.mSyncBranch,
300                                                    "extended" + i,
301                                                    this.mTypes.CHAR));
302   }
