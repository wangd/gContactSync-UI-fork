 41 /**
 42  * A simple class to store information about a preference (label, type, default
 43  * value and the actual value).  Set the value after creating the pref.
 44  * @param aName    {string} The label (or name) of the preference.
 45  * @param aType    {string} The type of preference (bool, int, or char).
 46  * @param aDefault {object} The default value for the preference.
 47  * @constructor
 48  * @class
 49  */
 50 com.gContactSync.Pref = function gCS_Pref(aName, aType, aDefault) {
 51   this.label        = aName;
 52   this.type         = aType;
 53   this.defaultValue = aDefault;
 54   this.value        = null;
 55 }
