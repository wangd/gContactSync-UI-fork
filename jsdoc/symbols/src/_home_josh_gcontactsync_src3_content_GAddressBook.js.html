 37 if (!com) var com = {}; // A generic wrapper variable
 38 // A wrapper for all GCS functions and variables
 39 if (!com.gContactSync) com.gContactSync = {};

 41 /**
 42 * An extension of AddressBook that adds functionality specific to gContactSync.
 43 * @param aDirectory {nsIAbDirectory} The actual directory.
 44 * @param aNoPrefs   {boolean}        Set this to true to skip fetching the
 45 *                                    preferences.
 46 * @constructor
 47 * @class
 48 * @extends com.gContactSync.AddressBook
 49 */
 50 com.gContactSync.GAddressBook = function gCS_GAddressBook(aDirectory, aNoPrefs) {
 51   // call the AddressBook constructor using this object
 52   com.gContactSync.AddressBook.call(this, aDirectory);
 53 
 54   // Preferences for this address book
 55   // If these aren't set the global preference with the same name, if any, is used
 56   // NOTE: All these preferences are converted to strings
 57   this.mPrefs = {
 58     Plugin:         "", // The name of the plugin to use
 59     Username:       "", // The username of the acct synced with
 60     Disabled:       "", // Temporarily disable synchronization with this AB
 61     // NOTE: These three prefs aren't combined into a single pref for backwards
 62     // compatibility with 0.2.x
 63     myContacts:     "", // true if only one group should be synced
 64     myContactsName: "", // The name of the group to sync
 65     syncGroups:     "", // Synchronize groups
 66     // NOTE: These two prefs aren't combined into a single pref for backwards
 67     // compatibility with 0.2.x
 68     readOnly:       "", // Fetch updates from Google but don't send any changes
 69     writeOnly:      "", // Send changes to the server, but don't fetch any changes
 70     updateGoogleInConflicts: "", // If a contact was updated in Google and TB then
 71                                  // this pref determines which contact to update
 72     lastSync:       "", // The last time this AB was synchronized
 73     lastBackup:     "", // The last time this AB was backed up
 74     reset:          ""  // Whether this AB has been reset since the last sync
 75   };
 76   if (!aNoPrefs)
 77     this.getPrefs();
 78 };
 79 
 80 // Copy the AB prototype (methods and member variables)
 81 com.gContactSync.GAddressBook.prototype = com.gContactSync.AddressBook.prototype;
 82 
 83 // A prefix for all preferences used to prevent conflicts with other extensions
 84 com.gContactSync.GAddressBook.prototype.prefPrefix = "gContactSync";
 85 
 86 /**
 87  * Fetches all of this directory's preferences.  If the directory does not have
 88  * any given preferences this function will use the global preference's value,
 89  * if any.
 90  */
 91 com.gContactSync.GAddressBook.prototype.getPrefs = function GAddressBook_getPrefs() {
 92   com.gContactSync.LOGGER.VERBOSE_LOG("\nGetting Prefs for AB '" + this.getName() + "':");
 93   var i, val, pref;
 94   for (i in this.mPrefs) {
 95     // all prefs except lastSync have the prefPrefix in from of them
 96     val = this.getStringPref((i === "lastSync" ? i : this.prefPrefix + i));
 97     // getStringPref returns 0 iff the pref doesn't exist
 98     // if the pref doesn't exist, then use the global gContactSync pref
 99     // AND set this AB's pref so this doesn't fall through next time
100     // this behavior is mostly for backwards compatibility
101     if (val === 0) {
102       com.gContactSync.LOGGER.VERBOSE_LOG("getPrefs fell through on " + i);
103       pref = com.gContactSync.Preferences.mSyncPrefs[i];
104       val  = pref ? String(pref.value) : "";
105       this.savePref(i, val);
106     }
107     com.gContactSync.LOGGER.VERBOSE_LOG(" * " + i + " = " + val);
108     this.mPrefs[i] = val;
109   }
110   com.gContactSync.LOGGER.VERBOSE_LOG("\n");
111 };
112 
113 /**
114  * Save the value of a given preference for this address book.
115  *
116  * @param aName  {string} The name of the preference to set.
117  * @param aValue {string} The value to set the preference to.
118  */
119 com.gContactSync.GAddressBook.prototype.savePref = function GAddressBook_savePref(aName, aValue) {
120   com.gContactSync.LOGGER.VERBOSE_LOG(" * Setting pref '" + aName + "' to value '" + aValue + "'");
121   // all prefs except lastSync have the prefPrefix in from of them
122   this.setStringPref((aName === "lastSync" ? aName : this.prefPrefix + aName), aValue);
123   // in theory (and in testing) the preferences listener should already take
124   // care of setting the preference in this.mPrefs...
125   this.mPrefs[aName] = aValue;
126 };
127  
128 /**
129  * Sets the last time this address book was synchronized, in milliseconds
130  * since the epoch.
131  * @param aLastSync {integer} The last sync time.
132  */
133 com.gContactSync.GAddressBook.prototype.setLastSyncDate = function GAddressBook_setLastSyncDate(aLastSync) {
134   this.setStringPref("lastSync", aLastSync);
135   // in theory (and in testing) the preferences listener should already take
136   // care of setting the preference in this.mPrefs...
137   this.mPrefs.lastSync = aLastSync;
138 };

140 /**
141  * 'Resets' this address book making it appear to be brand new and never
142  * synchronized.
143  * The username is NOT erased.
144  * 
145  * This includes:
146  *   - Creating a backup
147  *   - Deleting all mailing lists
148  *   - Deleting all contacts
149  *   - Setting primary to true
150  *   - Setting the last sync date to 0
151  * @returns {boolean} True if the AB was reset, false otherwise.
152  */
153 com.gContactSync.GAddressBook.prototype.reset = function GAddressBook_reset() {
154   com.gContactSync.LOGGER.LOG("Resetting the " + this.getName() + " directory.");
155   var lists, i, dt;
156   if (this.mPrefs.reset === "true") {
157     com.gContactSync.LOGGER.LOG_WARNING("An attempt was made to reset an AB which was already reset.  Ignoring request.");
158     return false;
159   }
160   dt = new Date().toLocaleFormat("%Y_%m_%d_");
161   com.gContactSync.GAbManager.backupAB(this, "reset_" + dt, ".bak");
162   try {
163     lists = this.getAllLists(true);
164   }
165   catch (e) {
166     com.gContactSync.LOGGER.LOG_ERROR("Unable to get all lists", e);
167     lists = {};
168   }
169   com.gContactSync.LOGGER.VERBOSE_LOG(" * Deleting all lists");
170   for (i in lists) {
171     if (lists[i] instanceof com.gContactSync.GMailList) {
172       com.gContactSync.LOGGER.VERBOSE_LOG("  - Deleting list " + lists[i].getName());
173       lists[i].remove();
174     }
175   }
176   com.gContactSync.LOGGER.VERBOSE_LOG(" * Finished deleting lists");
177   com.gContactSync.LOGGER.VERBOSE_LOG(" * Deleting all contacts");
178   this.deleteContacts(this.getAllContacts());
179   com.gContactSync.LOGGER.VERBOSE_LOG(" * Setting Last Sync Date to 0");
180   this.setLastSyncDate(0);
181   com.gContactSync.LOGGER.LOG("Finished resetting the directory.");
182   // mark the AB as having been reset
183   this.savePref("reset", true);
184   return true;
185 };
186 /**
187  * Returns a new GMailList object given the same parameters as the GMailList
188  * constructor.
189  *
190  * See the GMailList constructor for the most recent comments.
191  *
192  * @param aList {Ci.nsIAbDirectory}       The actual nsIAbDirectory
193  *                                        representation of a mailing list.
194  * @param aParentDirectory {GAddressBook} The parent directory (as an
195  *                                        AddressBook object) containing this
196  *                                        mailing list.
197  * @param aNew             {boolean}      Set as true for new mailing lists where
198  *                                        no attempt should be made to fetch the
199  *                                        contacts contained in the list.
200  * @returns {GMailList} A new GMailList.
201  */
202 com.gContactSync.GAddressBook.prototype.newListObj = function GAddressBook_newListObj(aList, aParentDirectory, aNew) {
203   return new com.gContactSync.GMailList(aList, aParentDirectory, aNew);
204 };
205 
206 /**
207  * Returns an an object containing GMailList objects whose attribute name is
208  * the name of the mail list.
209  * @param skipGetCards {boolean} True to skip getting the cards of each list.
210  * @returns {object} An object containing GMailList objects.
211  */
212 com.gContactSync.GAddressBook.prototype.getAllLists = function GAddressBook_getAllLists(skipGetCards) {
213   // same in Thunderbird 2 and 3
214   com.gContactSync.LOGGER.VERBOSE_LOG("Searching for mailing lists:");
215   var iter = this.mDirectory.childNodes,
216       obj  = {},
217       list,
218       id,
219       data;
220   while (iter.hasMoreElements()) {
221     data = iter.getNext();
222     if (data instanceof Components.interfaces.nsIAbDirectory && data.isMailList) {
223       list    = this.newListObj(data, this, skipGetCards);
224       id      = list.getGroupID();
225       obj[id] = list;
226       com.gContactSync.LOGGER.VERBOSE_LOG(" * " + list.getName() + " - " + id);
227     }
228   }
229   return obj;
230 };
