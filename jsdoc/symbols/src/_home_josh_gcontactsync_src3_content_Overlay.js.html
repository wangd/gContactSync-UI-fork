133   /**
134    * Adds treecol elements to the address book results tree that shows cards in
135    * the currently selected directory.  These treecols allow the user to show
136    * and sort by extra attributes that are added by this extension.  This will
137    * only work after Bug 413260 landed, so in Thunderbird 3.0b1pre and after.
138    */
139   addTreeCols: function Overlay_addTreeCols() {
140     // get the treecols XUL element
141     var treeCols = document.getElementById("abResultsTreeCols");
142     if (!treeCols || !treeCols.appendChild)
143       return;
144     if (com.gContactSync.Preferences.mSyncPrefs.phoneColLabels.value) {
145       // fix the existing phone numbers
146       var arr = ["WorkPhone", "HomePhone", "FaxNumber"," CellularNumber",
147                  "PagerNumber"
148                 ];
149       // the strings from the string bundle
150       //var arr2 = ["first", "second", "third", "fourth", "fifth"];
151       var elem;
152       for (var i = 0; i < arr.length; i++) {
153         elem = document.getElementById(arr[i]);
154         if (!elem) {
155           continue;
156         }
157         // remove it
158         treeCols.removeChild(elem);
159         elem.setAttribute("label", com.gContactSync.StringBundle.getStr(arr[i]));
160         // and then add it to the end of the treecols element
161         treeCols.appendChild(elem);
162       }
163     }
164     // if Bug 413260 isn't applied in this version of TB, or if the pref was
165     // changed to false, then stop here
166     if (!this.mBug413260 || !com.gContactSync.Preferences.mSyncPrefs.newColLabels.value)
167       return;
168     // get the added attributes
169     var ids = com.gContactSync.ContactConverter.getExtraSyncAttributes(false),
170         id, splitter, treeCol;
171     // iterate through every added attribute and add a treecol for it unless
172     // it is a postal address
173     for (i = 0, length = ids.length; i < length; i++) {
174       id = ids[i];
175       if (id.indexOf("Type") !== -1)
176         continue; // skip addresses and Types
177       // make and add the splitter first
178       splitter = document.createElement("splitter");
179       splitter.setAttribute("class", "tree-splitter");
180       treeCols.appendChild(splitter);
181       // make the new treecol
182       treeCol = document.createElement("treecol");
183       // then set it up with the ID and other attributes
184       treeCol.setAttribute("id", id);
185       treeCol.setAttribute("class", "sortDirectionIndicator");
186       treeCol.setAttribute("hidden", "true");
187       treeCol.setAttribute("persist", "hidden ordinal width sortDirection");
188       treeCol.setAttribute("flex", "1");
189       treeCol.setAttribute("label", com.gContactSync.StringBundle.getStr(id));
190       // append it to the treecols element
191       treeCols.appendChild(treeCol);
192     }

394   /**
395    * Checks to see whether or not there is an authentication token in the login
396    * manager.  If so, it begins a sync.  If not, it shows the login prompt.
397    */
398   checkAuthentication: function Overlay_checkAuthentication() {
399     if (com.gContactSync.gdata.isAuthValid()) {
400       if (this.mUsername) {
401         var name = com.gContactSync.Preferences.mSyncPrefs.addressBookName.value;
402         var ab   = new com.gContactSync.GAddressBook(com.gContactSync.GAbManager.getAbByName(name));
403         ab.savePref("Username", this.mUsername);
404         ab.setLastSyncDate(0);
405         com.gContactSync.Sync.begin();
406       }
407       else
408         com.gContactSync.Sync.schedule(com.gContactSync.Preferences.mSyncPrefs.initialDelay.value);  
409       return;
410     }
411     this.setStatusBarText(com.gContactSync.StringBundle.getStr("notAuth"));
412     this.promptLogin();
413   },
414   /**
415    * Prompts the user to enter his or her Google username and password and then
416    * gets an authentication token to store and use.
417    */
418   promptLogin: function Overlay_promptLogin() {
419     var prompt   = Components.classes["@mozilla.org/embedcomp/prompt-service;1"]
420                              .getService(Components.interfaces.nsIPromptService)
421                              .promptUsernameAndPassword;
422     var username = {};
423     var password = {};
424     // opens a username/password prompt
425     var ok = prompt(window, com.gContactSync.StringBundle.getStr("loginTitle"),
426                     com.gContactSync.StringBundle.getStr("loginText"), username, password, null,
427                     {value: false});
428     if (!ok)
429       return false;
430 
431     // This is a primitive way of validating an e-mail address, but Google takes
432     // care of the rest.  It seems to allow getting an auth token w/ only the
433     // username, but returns an error when trying to do anything w/ that token
434     // so this makes sure it is a full e-mail address.
435     if (username.value.indexOf("@") < 1) {
436       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("invalidEmail"));
437       return this.promptLogin();
438     }
439     
440     // fix the username before authenticating
441     username.value = com.gContactSync.fixUsername(username.value);
442     var body     = com.gContactSync.gdata.makeAuthBody(username.value, password.value);
443     var httpReq  = new com.gContactSync.GHttpRequest("authenticate", null, null, body);
444     // if it succeeds and Google returns the auth token, store it and then start
445     // a new sync
446     httpReq.mOnSuccess = function authSuccess(httpReq) {
447       com.gContactSync.Overlay.login(username.value,
448                                      httpReq.responseText.split("\n")[2]);
449     };
450     // if it fails, alert the user and prompt them to try again
451     httpReq.mOnError = function authError(httpReq) {
452       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr('authErr'));
453       com.gContactSync.LOGGER.LOG_ERROR('Authentication Error - ' +
454                                         httpReq.status,
455                                         httpReq.responseText);
456       com.gContactSync.Overlay.promptLogin();
457     };
458     // if the user is offline, alert them and quit
459     httpReq.mOnOffline = com.gContactSync.Sync.mOfflineFunction;
460     httpReq.send();
461     return true;
462   },

475     // when the setup window loads, set its onunload property to begin a sync
476     setup.onload = function onloadListener() {
477       setup.onunload = function onunloadListener() {
478         com.gContactSync.Overlay.checkAuthentication();
479       };
480     };

683   /**
684    * A helper method for myDisplayCardViewPane that iterates through an array of
685    * attributes and returns true if at least one of them is present in the given
686    * card.
687    * @param aCard         {nsIAbCard} The card whose attributes are checked.
688    * @param aArray        {array}     The array of attributes to check for in
689    *                                  the card.
690    * @param aVisible      {boolean}   Optional. True if the element was
691    *                                  previously visible.
692    * @param aUseTypeLabel {boolean}   Optional.  True if the labels should be
693    *                                  the type of the attribute instead of the
694    *                                  attribute's name.
695    * @returns {boolean} True if at least one attribute in aArray is present in aCard.
696    */
697   getVisible: function Overlay_getVisible(aCard, aArray, aVisible, aUseTypeLabel) {
698     var visible = aVisible;
699     // return true if the card has the current attribute
700     for (var i = 0; i < aArray.length; i++) {
701       var attr = aArray[i];
702       var value = com.gContactSync.GAbManager.getCardValue(aCard, attr);
703       // get the name of the string to find in the bundle
704       var label = aUseTypeLabel ? com.gContactSync.GAbManager.getCardValue(aCard, attr + "Type")
705                                 : attr;
706       // get the actual string
707       // if the label is null (ie aUseTypeLabel was true, but there wasn't a type)
708       // then use the attribute's string as a default value
709       var str = label && label != "" ? com.gContactSync.StringBundle.getStr(label)
710                                      : com.gContactSync.StringBundle.getStr(attr);
711       var prefix = this.links[label];
712       // Make this a link if there is a prefix (aim:goim?...), the pref is set,
713       // and there is a box for this data
714       if (prefix && com.gContactSync.Preferences.mSyncPrefs.enableImUrls.value && cvData["cv" + attr + "Box"]) {
715         var suffix = this.links[label + "S"] ? this.links[label + "S"] : "";
716         visible    = HandleLink(cvData["cv" + attr], str, value,
717                                 cvData["cv" + attr + "Box"], prefix +
718                                 value + suffix) || visible;
719       }
720       else
721         visible = cvSetNodeWithLabel(cvData["cv" + attr], str, value) || visible;
722     }
723     return visible;
724   },
