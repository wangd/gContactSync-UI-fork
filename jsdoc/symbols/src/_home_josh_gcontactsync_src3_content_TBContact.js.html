 37 if (!com) var com = {}; // A generic wrapper variable
 38 // A wrapper for all GCS functions and variables
 39 if (!com.gContactSync) com.gContactSync = {};
 40 
 41 /**
 42  * Makes a new TBContact object that has functions to get and set various values
 43  * for a contact independently of the version of Thunderbird (using com.gContactSync.GAbManager).
 44  * Optionally takes the parent directory and is able to update the card in that
 45  * directory.
 46  * 
 47  * @param aContact   {nsIAbCard}   A Thunderbird contact.
 48  * @param aDirectory {AddressBook} The parent directory.  Optional.
 49  * @class
 50  * @constructor
 51  */
 52 com.gContactSync.TBContact = function gCS_TBContact(aCard, aDirectory) {
 53   if (!(aCard instanceof Components.interfaces.nsIAbCard)) {
 54     com.gContactSync.LOGGER.LOG_ERROR("Invalid aCard passed to the TBContact constructor: " +
 55                                       aCard + "\nCalled by: " + this.caller);
 56     throw "Invalid aCard passed to TBContact";
 57   }
 58   //if (!(aDirectory instanceof com.gContactSync.AddressBook)) {
 59   //  throw "Error - invalid directory sent to the TBContact constructor";
 60   //}
 61   this.mAddressBook = aDirectory;
 62   this.mContact     = aCard;
 63   this.mPostbox     = this.mContact.setAdditionalEmailAddresses;
 64 };
 65 
 66 com.gContactSync.TBContact.prototype = {
 67   /**
 68    * Returns the value of the requested property of this contact.
 69    *
 70    * If the readOnly preference is enabled, then this will return 0 for the
 71    * LastModifiedDate.
 72    * 
 73    * @param aAttribute {string} The attribute to get (PrimaryEmail, for example)
 74    *
 75    * @returns {string} The value of the attribute, or null if not set.
 76    */
 77   getValue: function TBContact_getValue(aAttribute) {
 78     if (!aAttribute)
 79       throw "Error - invalid attribute sent to TBContact.getValue";
 80     if (aAttribute === "LastModifiedDate" && this.mAddressBook &&
 81         this.mAddressBook.mPrefs && this.mAddressBook.mPrefs.readOnly === "true") {
 82       com.gContactSync.LOGGER.VERBOSE_LOG(" * Read only mode, setting LMD to 0");
 83       return 0;
 84     }
 85     // Postbox stores additional e-mail addresses already
 86     if (this.mPostbox && (aAttribute === "ThirdEmail" || aAttribute === "FourthEmail")) {
 87       var arrLen   = {},
 88           emailArr = this.mContact.getAdditionalEmailAddresses(arrLen);
 89       if (aAttribute === "ThirdEmail" && emailArr.length > 0) {
 90         return emailArr[0];
 91       }
 92       else if (emailArr.length > 1) {
 93         return emailArr[1];
 94       }
 95       return null;
 96     }
 97     return com.gContactSync.GAbManager.getCardValue(this.mContact, aAttribute);
 98   },
 99   /**
100    * Sets the value of the requested attribute of this contact and optionally
101    * updates the contact in its parent directory.
102    *
103    * @param aAttribute {string} The attribute to set (PrimaryEmail, for example)
104    * @param aValue     {string} The value for the given attribute.  If null the
105    *                   attribute is 'deleted' from the contact.
106    * @param aUpdate    {boolean} Set to true to update this card after setting
107    *                   the value of the attribute.
108    * @returns {boolean} True if the contact was updated.
109    */
110   setValue: function TBContact_setValue(aAttribute, aValue, aUpdate) {
111     if (this.mPostbox && (aAttribute === "ThirdEmail" || aAttribute === "FourthEmail")) {
112       // get the existing e-mail addresses
113       var arrLen   = {},
114           emailArr = this.mContact.getAdditionalEmailAddresses(arrLen);
115       if (aAttribute === "ThirdEmail") {
116         emailArr[0] = aValue;
117       }
118       // FourthEmail
119       else if (emailArr.length > 0) {
120         emailArr[1] = aValue;
121       }
122       else {
123         emailArr[0] = aValue;
124       }
125       this.mContact.setAdditionalEmailAddresses(emailArr.length, emailArr);
126     }
127     else {
128       com.gContactSync.GAbManager.setCardValue(this.mContact, aAttribute, aValue);
129     }
130     if (aUpdate)
131       return this.update();
132     return false;
133   },
134   /**
135    * Updates this card in its parent directory, if possible.
136    * @returns {boolean} True if the contact was updated.
137    */
138   update: function TBContact_update() {
139     if (!this.mAddressBook) {
140       com.gContactSync.LOGGER.LOG_WARNING("Warning - TBContact.update called w/o a directory");
141       return false;
142     }
143     return this.mAddressBook.updateContact(this);
144   },
145   /**
146    * Removes this card from its parent directory, if possible.
147    * @returns {boolean} True if the contact was removed.
148    */
149   remove: function TBContact_remove() {
150     if (!this.mAddressBook) {
151       com.gContactSync.LOGGER.LOG_WARNING("Warning - TBContact.remove called w/o a directory");
152       return false;
153     }
154     return this.mAddressBook.deleteContacts([this]);
155   },
156   /**
157    * Returns a 'name' for this contact.  It is the first non-null and not blank
158    * value for the following attributes:
159    *  - DisplayName
160    *  - PrimaryEmail
161    *  - GoogleID
162    * @returns {string} The name of this contact.
163    */
164   getName: function TBContact_getName() {
165     var displayName  = this.getValue("DisplayName");
166     if (displayName)
167       return displayName;
168     var primaryEmail = this.getValue("PrimaryEmail");
169     if (primaryEmail)
170       return primaryEmail;
171     return this.getValue("GoogleID");
172   },
173   /**
174    * Returns true if an only if this contact has a value for one or more of the
175    * postal address fields (Address, Address2, City, State, ZipCode, or Country)
176    * of the given type (Home, Work, etc.)
177    * @param aPrefix {string} The type of address (Home, Work, etc.)
178    * @return {boolean} True if and only if one or more of the address fields of
179    *                   the given type has a value.
180    */
181   hasAddress: function TBContact_hasAddress(aPrefix) {
182     return this.getValue(aPrefix + "Address") ||
183            this.getValue(aPrefix + "Address2") ||
184            this.getValue(aPrefix + "City") ||
185            this.getValue(aPrefix + "State") ||
186            this.getValue(aPrefix + "ZipCode") ||
187            this.getValue(aPrefix + "Country");
188   }
189 };
