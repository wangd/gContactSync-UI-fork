 48 /**
 49  * A class for reading, writing, and appending to files with an nsIFile for
 50  * storing data, authentication info, and logs.
 51  * @class
 52  */
 53 com.gContactSync.FileIO = {
 54   /** An nsIFile where the log is written */
 55   mLogFile: null,
 56   /** File names */
 57   fileNames: {
 58     /** The file where preferences are stored */
 59     PREFS_JS:    "prefs.js",
 60     /** Stores AB backups and the gContactSync log */
 61     FOLDER_NAME: "gcontactsync",
 62     /** Stores the log from the last sync */
 63     LOG_FILE:    "gcontactsync_log.txt"
 64   },
 65   /**
 66    * Initializes the files contained in this class.
 67    * Also creates the log directory, if necessary.
 68    */
 69   init: function FileIO_init() {
 70     var directory = this.getProfileDirectory();
 71     // Make sure the profile directory exists (if not something is very wrong)
 72     this.checkDirectory(directory);
 73     // Append the gcontactsync folder onto the directory and make sure it exists
 74     directory.append(this.fileNames.FOLDER_NAME);
 75     this.checkDirectory(directory);
 76     this.mLogFile   = directory;
 77     this.mLogFile.append(this.fileNames.LOG_FILE);
 78     if (this.mLogFile.exists() && !this.mLogFile.isWritable()) {
 79       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("logNotWritable") +
 80                                   "\n" + this.mLogFile.path);
 81       throw "Error - cannot write to the log file: " +
 82             com.gContactSync.FileIO.mLogFile.path;
 83     }
 84   },
 85   /**
 86    * Checks that a directory corresponding to an nsIFile exists, and creates it
 87    * if necessary.
 88    * This will throw an error if the directory could not be created, is not a
 89    * directory, or if the directory is not writeable.
 90    */
 91   checkDirectory: function FileIO_checkDirectory(aDirectory) {
 92     // If the directory doesn't exist yet then create it
 93     if (!aDirectory.exists()) {
 94       // create the directory (type = 1) - rw for the user and r for others
 95       try { aDirectory.create("1", parseInt("755", 8)); } catch (e) {}
 96       // if it still doesn't exist let the user know, then quit
 97       if (!aDirectory.exists()) {
 98         com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("couldntMkDir") +
 99                                     "\n" + aDirectory.path);
100         throw "Error - could not create the following directory: " +
101               aDirectory.path;
102       }
103     }
104     if (!aDirectory.isDirectory()) {
105       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("isNotDir") +
106                                   "\n" + aDirectory.path);
107       throw "Error - " + aDirectory.path + " is not a directory.";
108     }
109     if (!aDirectory.isWritable()) {
110       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("notWritable") +
111                                   "\n" + aDirectory.path);
112       throw "Error - Cannot write to the following directory: " +
113             aDirectory.path;
114     }
115   },
116   /**
117    * Gets the the file in the extension's directory with the given name.
118    * @param aName {string} The name of the file to get.
119    * @returns {nsIFile} The file with the given name in the extension's
120    *                   directory.
121    */
122   getFileInExtDir: function FileIO_getFileInExtDir(aName) {
123     var MY_ID = "gContactSync@pirules.net",
124         em    = Components.classes["@mozilla.org/extensions/manager;1"]
125                        .getService(Components.interfaces.nsIExtensionManager);
126     return em.getInstallLocation(MY_ID).getItemFile(MY_ID, aName);
127   },
128   /**
129    * Returns an nsIFile of the current profile directory of the application.
130    * @returns {nsIFile} The current profile directory of the application.
131    */
132   getProfileDirectory: function FileIO_getProfileDirectory() {
133     return Components.classes["@mozilla.org/file/directory_service;1"]
134                      .getService(Components.interfaces.nsIProperties)
135                      .get("ProfD", Components.interfaces.nsIFile);
136   },
137   /**
138    * Opens the given file and returns an array of the lines within it.
139    * @param aFile  {nsIFile} The nsIFile to read.
140    * @returns {array} An array of the lines in the file or [] if there is an
141    *                 error.
142    */
143   readFile: function FileIO_readFile(aFile) {
144     this.checkFile(aFile);
145     if (!aFile.exists())
146       return [];
147     try {
148       var line    = {},
149           lines   = [],
150           hasmore,
151           istream = Components.classes["@mozilla.org/network/file-input-stream;1"]
152                               .createInstance(Components.interfaces.nsIFileInputStream);
153       istream.init(aFile, 0x01, 0444, 0);
154       istream.QueryInterface(Components.interfaces.nsILineInputStream);
155 
156       do {
157         hasmore = istream.readLine(line);
158         lines.push(line.value);
159       } while (hasmore);
160 
161       istream.close();
162       return lines;
163     }
164     catch (e) {
165       throw "Unable to read from file: " + aFile + "\n" + e;
166     }
167   },
168   /**
169    * Writes the string data to the nsIFile aFile.
170    * NOTE: This will delete any existing text in the file.
171    * @param aFile  {nsIFile} The nsIFile to which the string is written.
172    * @param aData  {string}  The string of data to write to the file.
173    * @returns {boolean} True if there is no error.
174    */
175   writeToFile: function FileIO_writeToFile(aFile, aData) {
176     this.checkFile(aFile);
177     if (!aData)
178       return false;
179     try {
180       var foStream = Components.classes["@mozilla.org/network/file-output-stream;1"]
181                                .createInstance(Components.interfaces.nsIFileOutputStream);
182       foStream.init(aFile, 0x02 | 0x08 | 0x20, 0666, 0);
183       foStream.write(aData, aData.length);
184       foStream.close();
185       return true;
186     }
187     catch (e) {
188       throw "Unable to write '" + aData + "' to file: " + aFile + "\n" + e;
189     }
190     return false;
191   },
192   /**
193    * Appends the string aData to the nsIFile aFile.
194    * @param aFile {nsIFile} The nsIFile to which the string is appended.
195    * @param aData {string} The string of data to append to the file.
196    * @returns {boolean} True if there is no error.
197    */
198   appendToFile: function FileIO_appendToFile(aFile, aData) {
199     if (!aData)
200       return false;
201     this.checkFile(aFile);
202     try {
203       var foStream = Components.classes["@mozilla.org/network/file-output-stream;1"]
204                                .createInstance(Components.interfaces.nsIFileOutputStream);
205       if (aFile.exists())
206         foStream.init(aFile, 0x02 | 0x10, 0666, 0);
207       else
208         foStream.init(aFile, 0x02 | 0x08 | 0x20, 0666, 0);
209       foStream.write(aData, aData.length);
210       foStream.close();
211       return true;
212     }
213     catch (e) {
214       throw "Unable to append '" + aData + "' to file: " + aFile + + "\n" + aFile.path + "\n" + e;
215     }
216     return false;
217   },
218   /**
219    * Copies the contents of one file to another using the readFile and writeFile
220    * methods of this class.
221    * If there is an error reading the source file, or if one or more of the
222    * given files are invalid an error will be thrown.
223    * NOTE: The contents of the destination file are removed permanently.
224    *
225    * @param aSrc  {nsIFile} The source file.
226    * @param aDest {nsIFile} The destination file.  NOTE: the contents of this
227    *                        file before the copy will be permanently removed.
228    * @returns {boolean} True if the copy finished successfully
229    */
230   copyFile: function FileIO_copyFile(aSrc, aDest) {
231     this.checkFile(aSrc);
232     this.checkFile(aDest);
233     // read the file into an array
234     var lines = this.readFile(aSrc) || [];
235     // write the array into a file
236     this.writeToFile(aDest, lines.join("\n"));
237     return true;
238   },
239   /**
240    * Checks that an argument is not null, is an instance of nsIFile, and that,
241    * if it exists, that it is a file (not a directory).
242    * @param aFile   {nsIFile} The file to check.
243    * @param aCaller {string}  The name of the calling method.
244    */
245   checkFile: function FileIO_checkFile(aFile) {
246     if (!aFile || !aFile instanceof Components.interfaces.nsIFile || (aFile.exists() && !aFile.isFile()))
247       throw "Invalid File: " + aFile + " sent to the '" + this.checkFile.caller +
248             "' method" + com.gContactSync.StringBundle.getStr("pleaseReport");
249   }
250 };
