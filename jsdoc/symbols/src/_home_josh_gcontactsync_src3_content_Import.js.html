232   /**
233    * Step 1: Get an initial, unauthorized oauth_token and oauth_token_secret.
234    * This is done mostly on pirules.org which contains the consumer token and
235    * secret for various sources and signs the parameters.
236    * pirules.org returns the response from the source, usually of the form:
237    * oauth_token=1234&oauth_token_secret=5678
238    *
239    * @param aSource {string} The source from which the contacts are obtained,
240    *                         in lowercase, as supported by pirules.org.
241    */
242   step1: function Import_step1(aSource) {
243     var imp      = com.gContactSync.Import,
244         callback = aSource == "facebook" ? imp.step2b : imp.step2a;
245     if (imp.mStarted) {
246       // TODO warn the user and allow him or her to cancel
247     }
248     
249     // Reset mOAuth
250     imp.mOAuth.oauth_token        = "";
251     imp.mOAuth.oauth_token_secret = "";
252     imp.mOAuth.oauth_verifier     = "";
253     imp.mOAuth.access_token       = "";
254     imp.mOAuth.expires            = "";
255     
256     com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('startingImport'));
257     imp.mStarted = true;
258     imp.mSource  = aSource;
259     // get an oauth_token and oauth_token_secret
260     imp.httpReqWrapper("http://www.pirules.org/oauth/index2.php?quiet&silent&step=1&source=" +
261                        imp.mSource,
262                        callback);
263   },
264   /**
265    * Step 2a: The first of two substeps where the user is prompted for his or
266    * her credentials on the third-party website.
267    * In this substep, gContactSync gets the login URL from pirules.org with
268    * all it's parameters and the oauth_signature.
269    * This is done in step 1 for OAuth 2.0 (Facebook only at the moment).
270    */
271   step2a: function Import_step2a(httpReq) {
272     var imp = com.gContactSync.Import,
273         response = httpReq.responseText;
274     if (!response) {
275       com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importFailed'));
276       com.gContactSync.LOGGER.LOG_ERROR("***Import failed to get the auth tokens");
277       return;
278     }
279     com.gContactSync.LOGGER.LOG("***IMPORT: Step 1 finished:\nContents:\n" +
280                                 response);
281     // parse and store the parameters from step 1 (oauth_token &
282     // oauth_token_secret)
283     imp.storeResponse(response.replace("&", "&"));
284     imp.httpReqWrapper("http://www.pirules.org/oauth/index2.php?quiet&silent&step=2&source=" +
285                        imp.mSource +
286                        "&oauth_token=" + imp.mOAuth.oauth_token +
287                        "&oauth_token_secret=" + imp.mOAuth.oauth_token_secret,
288                        imp.step2b);
289   },
290   /**
291    * Step 2b: The second of two substeps where the user is prompted for his or
292    * her credentials on the third-party website.
293    * In this substep, gContactSync opens a browser to the login page for the
294    * particular source.
295    */
296   step2b: function Import_step2b(httpReq) {
297     var imp = com.gContactSync.Import,
298         response = httpReq.responseText;
299     if (!response) {
300       com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importFailed'));
301       com.gContactSync.LOGGER.LOG_ERROR("***Import failed to get the login URL");
302       return;
303     }
304     response = String(response).replace(/\&\;/g, "&");
305     com.gContactSync.LOGGER.LOG("***IMPORT: Step 2a finished:\nContents:\n" + response);
306     com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importRequestingCredentials'));
307     imp.openBrowserWindow(response, imp.logStep2b);
308   },

319   /**
320    * Step 3: Gets the new oauth_token then activates the token.
321    * This step must be initiated by the user (for now).
322    * TODO - find a way to automatically start step3 when possible.
323    */
324   step3: function Import_step3() {
325     var imp = com.gContactSync.Import;
326     if (!imp.mStarted) {
327       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("importNotStarted"));
328       return;
329     }
330     // Get the new oauth_token from the window.
331     imp.mOAuth.oauth_token = encodeURIComponent(imp.mWindow.document.getElementById('response').innerHTML);
332     // Get the oauth_verifier, if any
333     if (imp.mWindow.document.getElementById("oauth_verifier")) {
334       imp.mOAuth.oauth_verifier = encodeURIComponent(imp.mWindow.document.getElementById('oauth_verifier').innerHTML);
335     }
336     imp.mWindow.close();
337     if (!imp.mOAuth.oauth_token) {
338       com.gContactSync.alert(com.gContactSync.StringBundle.getStr('importCanceled'),
339                              com.gContactSync.StringBundle.getStr('importCanceledTitle'),
340                              window);
341       imp.mStarted = false;
342       return;
343     }
344     com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importActivatingToken'));
345     // activate the token
346     imp.httpReqWrapper("http://www.pirules.org/oauth/index2.php?quiet&silent&step=3&source=" +
347                          imp.mSource +
348                          "&oauth_token=" + imp.mOAuth.oauth_token +
349                          "&oauth_token_secret=" + imp.mOAuth.oauth_token_secret +
350                          (imp.mOAuth.oauth_verifier ? "&oauth_verifier=" + imp.mOAuth.oauth_verifier : ""),
351                          imp.step4);
352   },
353   /**
354    * Step 4: Use the token to fetch the user's contacts.
355    * This sends a request and the token/token secret to pirules.org which
356    * signs and sends the request to the source's @me/@friend URL.
357    */
358   step4: function Import_step4(httpReq) {
359     var imp = com.gContactSync.Import,
360         response = httpReq.responseText;
361     if (!response) {
362       com.gContactSync.LOGGER.LOG("***Import failed on step 3");
363       return;
364     }
365     com.gContactSync.LOGGER.LOG("***IMPORT: Step 3 finished:\nContents:\n" + response);
366     imp.storeResponse(response.replace("&", "&"));
367     com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importRetrievingContacts'));
368     // Use the token to fetch the user's contacts
369     // access_token is used instead of the oauth_token in OAuth 2.0
370     if (imp.mOAuth.access_token) {
371       imp.httpReqWrapper("http://www.pirules.org/oauth/index2.php?quiet&silent&step=4&source=" +
372                          imp.mSource +
373                          "&access_token=" + imp.mOAuth.access_token,
374                          imp.finish);
375     }
376     else {
377       imp.httpReqWrapper("http://www.pirules.org/oauth/index2.php?quiet&silent&step=4&source=" +
378                          imp.mSource +
379                          "&oauth_token=" + imp.mOAuth.oauth_token +
380                          "&oauth_token_secret=" + imp.mOAuth.oauth_token_secret,
381                          imp.finish);
382     }
383   },

402   /**
403    * Parses and stores a URL-encoded response in the following format:
404    * param1=value1&param2=value2&param3=value3...
405    * The parsed parameters and values are stored (still encoded) in
406    * com.gContactSync.Import.mOAuth[param] = value;
407    *
408    * @param aResponse {string} The encoded response to parse.
409    */
410   storeResponse: function Import_storeResponse(aResponse) {
411     var imp    = com.gContactSync.Import,
412         params = (aResponse).split("&");
413     for (var i = 0; i < params.length; i++) {
414       var index = params[i].indexOf("=");
415       if (index > 0) {
416         var param = params[i].substr(0, index),
417             value = params[i].substr(index + 1);
418         com.gContactSync.LOGGER.VERBOSE_LOG("***" + param + "=>" + value);
419         imp.mOAuth[param] = value;
420       }
421     }

442   /**
443    * Begins the actual import given a JSON feed of contacts.
444    * It promps the user for a name for the destination AB (can be new or old).
445    *
446    * @param aFeed {string} The JSON feed of contacts to parse.
447    */
448   beginImport: function Import_beginImport(aFeed) {
449     if (!aFeed) {
450       return;
451     }
452     try {
453       com.gContactSync.LOGGER.VERBOSE_LOG(aFeed);
454       // decode the JSON and get the array of cards
455       var nsIJSON = Components.classes["@mozilla.org/dom/json;1"]
456                               .createInstance(Components.interfaces.nsIJSON);
457       try {
458         var obj = nsIJSON.decode(aFeed);
459       }
460       catch (e) {
461         com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("importFailedMsg"));
462         com.gContactSync.LOGGER.LOG_ERROR("Import failed: ", aFeed);
463         com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importFailed'));
464         return;
465       }
466       var res = com.gContactSync.prompt(com.gContactSync.StringBundle.getStr("importDestination"),
467                                         com.gContactSync.StringBundle.getStr("importDestinationTitle"),
468                                         window);
469       if (!res) {
470         return;
471       }
472       var ab = new com.gContactSync.GAddressBook(com.gContactSync.GAbManager.getAbByName(res),
473                                                  true);
474       var arr = obj.entry || obj.data;
475       for (var i in arr) {
476         var contact = arr[i],
477             id = contact.id;
478         if (id) {
479           var newCard = ab.newContact(),
480               attr    = "";
481           // Download FB photos
482           if (this.mSource === "facebook") {
483             var file = com.gContactSync.writePhoto("https://graph.facebook.com/" + id + "/picture?type=large",
484                                                    id + "_" + (new Date()).getTime());
485             if (file) {
486               com.gContactSync.LOGGER.VERBOSE_LOG("Wrote photo...name: " + file.leafName);
487               newCard.setValue("PhotoName", file.leafName);
488               newCard.setValue("PhotoType", "file");
489               newCard.setValue("PhotoURI",
490                                Components.classes["@mozilla.org/network/io-service;1"]
491                                          .getService(Components.interfaces.nsIIOService)
492                                          .newFileURI(file)
493                                          .spec);
494             }
495           }
496           // Iterate through each attribute in the JSON contact
497           for (var j in contact) {
498             // If there is a map for just this source, check it for the
499             // attribute first, otherwise just use the default map.
500             if (this["mMap" + this.mSource])
501               attr = this["mMap" + this.mSource][j] || this.mMap[j];
502             else
503               attr = this.mMap[j];
504             if (attr) {
505               // Download a photo of the user, if available.
506               if (j === "picture" || j === "thumbnailUrl" || j === "photos") {
507                 var file = com.gContactSync.writePhoto((j === "photos" ? contact[j][0].value : contact[j]),
508                                                        this.mSource + "_" + id,
509                                                        0);
510                 if (file) {
511                   com.gContactSync.LOGGER.VERBOSE_LOG("Wrote photo...name: " + file.leafName);
512                   newCard.setValue("PhotoName", file.leafName);
513                   newCard.setValue("PhotoType", "file");
514                   newCard.setValue("PhotoURI",
515                                    Components.classes["@mozilla.org/network/io-service;1"]
516                                              .getService(Components.interfaces.nsIIOService)
517                                              .newFileURI(file)
518                                              .spec);
519                 }
520               }
521               // when contact[j] is an Array things are a bit more
522               // complicated
523               else if (contact[j] instanceof Array) {
524                 // emails: [
525                 //   {email: somebody@somwhere, type: work},
526                 //   {email: somebody2@somwhere, type: work}
527                 // ]
528                 // contact[j]    = emails[]
529                 // contact[j][k] = emails[k]
530                 for (var k = 0; k < contact[j].length; k++) {
531                   if (!attr[k]) break;
532                   // contact[j][k][l] = sombody@somewhere
533                   for (var l in contact[j][k]) {
534                     if (attr[l]) {
535                       var type = contact[j][k].type;
536                       // not all arrays can be mapped to TB fields by index
537                       // TODO - support using original phone # fields
538                       // this would require NOT storing the type...
539                       var tbAttribute = String(attr[k] + attr[l]).replace("<type>", type);
540                       // Workaround for inconsistent phone number attributes in TB
541                       if (attr === "phoneNumbers" && (type === "Cellular" || type === "Pager" || type === "Fax")) {
542                         tbAttribute = tbAttribute.replace("Phone", "Number");
543                       }
544                       // mMap[j][[k] is the prefix (Primary, Second, etc.)
545                       // mMap[j][l] is the suffix (Email)
546                       com.gContactSync.LOGGER.VERBOSE_LOG(" - (Array): " + tbAttribute + "=" + contact[j][k][l]);
547                       newCard.setValue(tbAttribute, this.decode(contact[j][k][l]));
548                     }
549                   }
550                   
551                 }
552               }
553               else if (j === "photos") {
554                 // TODO download the photo...
555                 // possibly implementation-specific
556               }
557               // if it is just a normal property (has a length property =>
558               // string) check the map
559               else if (attr.length) {
560                 com.gContactSync.LOGGER.VERBOSE_LOG(" - (String): " + attr + "=" + contact[j])
561                 newCard.setValue(attr, this.decode(contact[j]));
562               }
563               // else it is an object with subproperties
564               else {
565                 for (var k in contact[j]) {
566                   if (attr[k]) {
567                     com.gContactSync.LOGGER.VERBOSE_LOG(" - (Object): " + attr[k] + "/" + j + "=" + contact[j][k]);
568                     newCard.setValue(attr[k], this.decode(contact[j][k]));
569                   }
570                 }
571               }
572             }
573           }
574           newCard.update();
575         }
576       }
577     }
578     catch (e) {
579       com.gContactSync.alertError(e);
580       return;
581     }
582     // refresh the ab results pane
583     try {
584       if (SetAbView !== undefined) {
585         SetAbView(GetSelectedDirectory(), false);
586       }
587       
588       // select the first card, if any
589       if (gAbView && gAbView.getCardFromRow(0))
590         SelectFirstCard();
591     }
592     catch (e) {}
593     com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importFinished'));
594     com.gContactSync.alert(com.gContactSync.StringBundle.getStr("importComplete"),
595                            com.gContactSync.StringBundle.getStr("importCompleteTitle"),
596                            window);
597   },
598   /**
599    * Decodes text returned in a JSON feed.
600    * @param aString {string} The text to decode.
601    * @returns {string} The decoded text.
602    */
603   decode: function Import_decode(aString) {
604     return aString ?
605             decodeURIComponent(aString).replace(/</g,   "<")
606                                        .replace(/>/g,   ">")
607                                        .replace(/&/g,  "&")
608                                        .replace(/"/g, '"') :
609             "";
610   },
611   /**
612    * A wrapper for HttpRequest for use when importing contacts.
613    * @param aURL {string} The URL to send the GET request to.
614    * @param aCallback {function} The callback function if the request succeeds.
615    */
616   httpReqWrapper: function Import_httpReqWrapper(aURL, aCallback) {
617     var httpReq   = new com.gContactSync.HttpRequest();
618     httpReq.mUrl  = aURL;
619     httpReq.mType = "GET";
620     httpReq.mOnSuccess = aCallback;
621     httpReq.mOnOffline = this.mOfflineFunction;
622     httpReq.mOnError = function import_onError(httpReq) {
623       com.gContactSync.alertError(com.gContactSync.StringBundle.getStr("importFailedMsg"));
624       com.gContactSync.LOGGER.LOG_ERROR("Import failed: ", httpReq.responseText);
625       com.gContactSync.Overlay.setStatusBarText(com.gContactSync.StringBundle.getStr('importFailed'));
626     }
627     httpReq.send();
628   },
